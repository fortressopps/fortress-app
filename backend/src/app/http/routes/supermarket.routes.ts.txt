// backend/src/app/http/routes/supermarket.routes.ts
import { Hono } from "hono";
import { supermarketController } from "../../modules/supermarket/supermarket.controller";

const router = new Hono();

// Healthcheck da feature
router.get("/health", (c) => c.json({ ok: true, feature: "supermarket" }));

router.get("/list", supermarketController.getList);
router.post("/list", supermarketController.createList);
router.get("/list/:id", supermarketController.getListById);
router.put("/list/:id", supermarketController.updateList);
router.delete("/list/:id", supermarketController.deleteList);

router.post("/item", supermarketController.createItem);
router.put("/item/:id", supermarketController.updateItem);
router.delete("/item/:id", supermarketController.deleteItem);
router.delete("/items/by-list/:listId", supermarketController.deleteItemsByList);

export default router;
